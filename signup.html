    <!-- Load the SDK -->
<script src="https://testdelegate.com/sdk/delegate.js"></script>

<script>
  // Initialize as soon as possible
  Delegate.init({ apiKey: 'your_key_here' });

  // On form submit, verify before continuing
  form.addEventListener('submit', async (e) => {
    e.preventDefault();
    const result = await Delegate.verify();

    if (result.verdict === 'block') {
      // Don't create the account
    } else if (result.verdict === 'challenge') {
      // Require extra verification (OTP, captcha, KYC)
    } else {
      // Allow the signup through
    }
  });
</script>
